func (r *TechnologyRepo) FindPopular(ctx context.Context, limit int) ([]domain.TechnologyListItem, error) {
    query := `
        SELECT 
            t.id,
            t.slug,
            t.name,
            t.description_short,
            t.readiness_level,
            t.image_url,
            tr.slug AS trend_slug,
            tr.name AS trend_name
        FROM technologies t
        JOIN trends tr ON t.trend_id = tr.id
        ORDER BY t.readiness_level DESC, t.created_at DESC
        LIMIT $1
    `
    
    rows, err := r.db.Query(ctx, query, limit)
    if err != nil {
        return nil, fmt.Errorf("query popular technologies: %w", err)
    }
    defer rows.Close()
    
    var items []domain.TechnologyListItem
    for rows.Next() {

            &item.ID,

            &item.Name,
            &item.DescriptionShort,
            &item.TRL,
            &item.ImageURL,
            &item.TrendSlug,
            &item.TrendName,
        )
        if err != nil {
            return nil, fmt.Errorf("scan row: %w", err)
        }
        items = append(items, item)

type CatalogService struct {
    repo  ports.CatalogRepository
    cache *cache.Cache  // ← ДОБАВИЛИ
}

func NewCatalogService(repo ports.CatalogRepository) *CatalogService {
    return &CatalogService{
        repo:  repo,
        cache: cache.New(5*time.Minute, 10*time.Minute),  // ← ДОБАВИЛИ
    }
}

func (s *CatalogService) ListTrends(ctx context.Context) ([]domain.Trend, error) {
    // Пытаемся достать из кэша
    if cached, found := s.cache.Get("trends"); found {
        return cached.([]domain.Trend), nil
    }
    
    // Если в кэше нет, идем в БД
    trends, err := s.repo.FindAllTrends(ctx)
    if err != nil {
        return nil, err
    }
    
    // Сохраняем в кэш
    s.cache.Set("trends", trends, cache.DefaultExpiration)
    
    return trends, nil
}
```

package httpapi

import (
    "net/http"
    "time"
    "go.uber.org/zap"
)

func LoggerMiddleware(logger *zap.Logger) func(http.Handler) http.Handler {
    return func(next http.Handler) http.Handler {
        return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
            start := time.Now()
            
            // Оборачиваем ResponseWriter чтобы захватить status code
            ww := &responseWriter{ResponseWriter: w, statusCode: http.StatusOK}
            
            next.ServeHTTP(ww, r)
            
            duration := time.Since(start)
            
            logger.Info("http request",
                zap.String("method", r.Method),
                zap.String("path", r.URL.Path),
                zap.Int("status", ww.statusCode),
                zap.Duration("duration", duration),
                zap.String("ip", r.RemoteAddr),
            )
        })
    }
}

type responseWriter struct {
    http.ResponseWriter
    statusCode int
}

func (rw *responseWriter) WriteHeader(code int) {
    rw.statusCode = code
    rw.ResponseWriter.WriteHeader(code)
}
```

### Пример 1: Транзакция в repository

```go
func (r *AdminTechnologyRepo) CreateWithTags(ctx context.Context, req domain.TechnologyCreateRequest, tagIDs []string) (*domain.Technology, error) {
    // Начинаем транзакцию
    tx, err := r.db.Begin(ctx)
    if err != nil {
        return nil, fmt.Errorf("begin tx: %w", err)
    }
    defer tx.Rollback(ctx)  // откатим если что-то пошло не так
    
    // 1. Создаем технологию
    query1 := `INSERT INTO technologies (...) VALUES (...) RETURNING id, ...`
    var tech domain.Technology
    err = tx.QueryRow(ctx, query1, req.Slug, req.Name, ...).Scan(&tech.ID, &tech.Slug, ...)
    if err != nil {
        return nil, fmt.Errorf("create technology: %w", err)
    }
    
    // 2. Добавляем теги
    for _, tagID := range tagIDs {
        query2 := `INSERT INTO technology_tags (technology_id, tag_id) VALUES ($1, $2)`
        _, err = tx.Exec(ctx, query2, tech.ID, tagID)
        if err != nil {
            return nil, fmt.Errorf("add tag %s: %w", tagID, err)
        }
    }
    
    // Коммитим транзакцию
    if err := tx.Commit(ctx); err != nil {
        return nil, fmt.Errorf("commit tx: %w", err)
    }
    
    return &tech, nil
}
```

func AdminRoleMiddleware(next http.Handler) http.Handler {

        }
        
        // Проверяем роль
        if user != "admin" {
            writeError(w, http.StatusForbidden, "forbidden: admin only")
            return
        }
        
        next.ServeHTTP(w, r)
    })
}
